# Go练习





### 定义两个变量交换变量的值

注：使用中间变量的方法

```go
package main

import "fmt"

func main() {

	// 交换变量a和变量b的值
	// 方法一:定义中间变量的方式
	a := 18
	b := 39
	c := a
	a = b
	b = c
	fmt.Println("交换后a的值:", a, "交换后b的值:", b)
	/*
	    输出效果
		交换后a的值: 39 交换后b的值: 18
	*/
}
```



### 定义两个变量交换变量的值

注：使用运算符方法



```go
package main

import "fmt"

func main() {

	// 交换变量a和变量b的值
	// 方法二:使用相减的方法
	var a = 18
	var b = 39
	var c = a + b
	fmt.Printf("运算符方法交换后a的值:%v,运算符方法交换后b的值是:%v", c-a, c-b)
	/*
		输出效果
		运算符方法交换后a的值:39,运算符方法交换后b的值是:18
	*/
}
```



### 运用for循环输出长方形



```go
package main

import "fmt"

func main() {
	// 利用for循环输出长方形
	var sec = 4
	var aer = 3
	for i := 0; i <= sec; i++ {
		for j := i; j <= aer; j++ {
			fmt.Print("#")
		}
		fmt.Println()
	}

}

```



### 运用for循环输出直接三角形



```go
package main

import "fmt"

func main() {
	// 利用for循环输出长方形
	var sec = 4
	var aer = 3
	for i := 0; i <= sec; i++ {
		for j := i - 1; j <= aer; j++ {
			fmt.Print("#")
		}
		fmt.Println()
	}
	/*
	#####
	####
	###
	##
	#
	*/
}
```



### 运用for输出九九乘法表



```go
package main

import "fmt"

func main() {
	// 利用for循环输出长方形
	var sec = 9
	var aer = 9
	for i := 1; i <= sec; i++ {
		for j := i; j <= aer; j++ {
			var product = i * j
			fmt.Printf("%v x %v = %v\t", i, j, product)
		}
		fmt.Println()
	}
```
