# Go-function05



### defer语句



```go
Go语言中的defer语句会将其后跟随的语句进行延迟处理，在defer归属的函数即将返回时，将延迟处理的语句按defer定义的逆序进行执行，也就是说，先被defer的语句最后被执行，最后被defer的语句，最先被执行。
```



### defer语句实例



```go
package main

import "fmt"

func main() {
	fmt.Println("开始")
	defer fmt.Println(1)
	defer fmt.Println(2)
	defer fmt.Println(3)
	fmt.Println("结束")
}
// 输出结果
/*
开始
结束
3
2
1
*/
```

### defer语句在命名返回值和匿名返回函数中的表现不一样



```go
package main

import "fmt"

// 匿名返回值
func Fn1() int {
	var a int
	defer func() {
		a++
	}()
	return a
}

// 命名返回值
func Fn2() (n int) {
	defer func() {
		n++
	}()
	return n
}

func main() {
	fmt.Printf("匿名返回值:%v\n", Fn1())
	fmt.Printf("命名返回值:%v\n", Fn2())

}
// 输出结果
/*
匿名返回值:0
命名返回值:1
*/
```



### defer注册要延迟执行的函数时该函数所有的参数都需要确定其值



```go
package main

import "fmt"

// defer语句函数返回匿名变量
func Fn01() int {
	var n = 6
	defer func() {
		n++
	}()
	return n
}

// defer语句函数返回命名变量
func Fn02() (n int) {
	defer func() {
		n++
	}()
	return 6
}

func Fn03() (n int) {
	m := 6
	defer func() {
		m++
	}()
	return m
}

func Fn04() (n int) {
	defer func(n int) {
		n++
	}(n)
	return 6
}

func main() {
	fmt.Printf("Fn01的值是:%v\n", Fn01())
	fmt.Printf("Fn02的值是:%v\n", Fn02())
	fmt.Printf("Fn03的值是:%v\n", Fn03())
	fmt.Printf("Fn04的值是:%v\n", Fn04())

}
// 输出结果
/*
Fn01的值是:6
Fn02的值是:7
Fn03的值是:6
Fn04的值是:6
*/
```



### defer注册和执行顺序



```go
package main

import "fmt"

func calc(index string, a, b int) int {
	ret := a + b
	fmt.Printf("index ===>%v  a===>%v  b===>%v\n", index, a, b)
	return ret
}

// defer语句注册顺序和执行顺序
func main() {
	x := 1
	y := 2
	defer calc("AAA", x, calc("a", x, y))
	y = 20
	defer calc("BBB", x, calc("b", x, y))

}
// 输出结果
/*
index ===>a  a===>1  b===>2
index ===>b  a===>1  b===>20
index ===>BBB  a===>1  b===>21
index ===>AAA  a===>1  b===>3
*/
```



### panic/recover



```go
package main

import "fmt"

// panic/recover

func Fn01() {
	fmt.Println("函数Fn01输出开始")
}

func Run01() {
	defer func() {
		err := recover()
		if err != nil {
			fmt.Println("程序运行正常")

		}
	}()
	panic("输出一个异常")
}

func main() {

	Fn01()
	Run01()
	fmt.Println("输出结束")

}
// 输出结果
/*
函数Fn01输出开始
程序运行正常
输出结束
*/
```



### recover输出异常错误信息



```go
package main

import (
	"fmt"
)

// panic/recover 抛出异常不影响程序执行

func division(a, b int) int {
	defer func() {
		errors := recover()
		if errors != nil {
			fmt.Printf("输出错误信息:%v", errors)
		}
	}()
	return a / b
}
func main() {
	div := division(10, 0)
	fmt.Printf("除法运算的值:%v", div)

}
// 输出结果
/*
输出错误信息:runtime error: integer divide by zero除法运算的值:0
*/
```



### 输出错误日志



```go
package main

import (
	"errors"
	"fmt"
)

// 错误日志输出
func readFile(fileName string) error {
	if fileName == "main.go" {
		return nil
	} else {
		return errors.New("传入文件名错误，读取失败...")
	}
}

func myFn() {
	defer func() {
		err := recover()
		if err != nil {
			fmt.Println("给管理员发送邮件，传入的文件错误...")
		}
	}()

	file := readFile("mai.go")
	if file != nil {
		panic(file)
	}
}

func main() {
	myFn()
	fmt.Println("程序执行完成")

}
// 输出结果
/*
给管理员发送邮件，传入的文件错误...
程序执行完成
*/
```
