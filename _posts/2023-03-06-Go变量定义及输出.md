### Println Print区别



> 1.Println输出时默认后面带换行符\n,Print输出默认后面是没有换行符\n,所以多行打印输出为一行；
>
> 2.Println打印输出多行时有空格，Print打印输出多行时无空格；



```go
// 没有区别的输出
package main

import "fmt"

func main() {
	fmt.Println("Println hello world")
	fmt.Print("Print hello world")

	/*
			输出效果
			Println hello world
		    Print hello world
	*/
}
// 区别 Println 换行，Print不行
package main

import "fmt"

func main() {
	// Println 和 Print区别

	fmt.Println("Println hello world")
	fmt.Println("Println hello world")
	fmt.Println("Println hello world")

	fmt.Print("Print hello world")
	fmt.Print("Print hello world")
	fmt.Print("Print hello world")

	/*
		输出结果
		Println hello world
		Println hello world
		Println hello world
		Print hello worldPrint hello worldPrint hello world
	*/

}
// 更直观的区分Println和Print
package main

import "fmt"

func main() {
	// Println 和 Print区别
	fmt.Println("---使用Println输出---")
	fmt.Println("A")
	fmt.Println("B")
	fmt.Println("C")
	fmt.Println("---使用Print输出---")
	fmt.Print("A")
	fmt.Print("B")
	fmt.Print("C")
	/*
		---使用Println输出---
		A
		B
		C
		---使用Print输出---
		ABC
	*/
}
// 一次输出多个字符
package main

import "fmt"

func main() {
	// Println 和 Print区别
	fmt.Println("---使用Println输出---")
	fmt.Println("A", "B", "C")
	fmt.Println("---使用Print输出---")
	fmt.Print("A", "B", "C")
	/*
		---使用Println输出---
	    A B C
	    ---使用Print输出---
	    ABC
	*/
}
```



### Println Printf区别



> 1.Go中变量定义后必须使用；
>
> 2.Printf 格式化输出；
>
> 3.:=方式定义变量



```go
// Println 和 Printf输出
package main

import "fmt"

func main() {
	// Println和Printf区别
	var a = "aaaaaaaa"
	fmt.Println("---使用Println打印输出---")
	fmt.Println(a)
	fmt.Println("---使用Printf打印输出---")
	fmt.Printf("变量a的值是:%v", a)

}
// Println 和 Printf输出对比
package main

import "fmt"

func main() {
	// Println和Printf区别
	var a = 10
	var b = 8
	var c = 6
	fmt.Println("---Println输出---")
	fmt.Println("a=", a, "b=", b, "c=", c)
	fmt.Println("---Printf输出---")
	fmt.Printf("a=%v,b=%v,c=%v", a, b, c)
	/*
		---Println输出---
		a= 10 b= 8 c= 6
		---Printf输出---
		a=10,b=8,c=6
	*/
}
// 类型推导法定义变量及%T输出变量的类型
package main

import "fmt"

func main() {
	// Println和Printf区别
	a := 60
	b := 80
	c := 90
	fmt.Println("使用格式化方式输出值")
	fmt.Printf("a=%v,b=%v,c=%v\n", a, b, c)
	fmt.Println("输出a,b,c的类型")
	fmt.Printf("a的类型:%T,b的类型:%T,c的类型:%T", a, b, c)
	/*
		使用格式化方式输出值
	    a=60,b=80,c=90
	    输出a,b,c的类型
	    a的类型:int,b的类型:int,c的类型:int
	*/

}
```
