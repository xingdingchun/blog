### 递归函数实现1-100的和



```go
package main

import "fmt"

// 函数调用函数
func Fn01() {
	fmt.Println("方法01")
}

func Fn02() {
	Fn01()
	fmt.Println("方法02")
}

// 函数实现输出数字
func Fn03(x int) {
	x--
	if x > 0 {
		fmt.Println(x)
		Fn03(x)
	}
}

// 使用函数递归的方法实现1-100的和
func Fn04(n int) int {
	if n > 1 {
		return n + Fn04(n-1)
	} else {
		return 1
	}
}

// 使用递归函数的方法实现5的阶乘
func Fn05(y int) int {
	if y > 1 {
		return y * Fn05(y-1)
	} else {
		return 1
	}
}

func main() {
	// 用for循环实现1-100的和
	var sum int
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Printf("使用for循环实现1-100的和:%v\n", sum)

	// 函数执行函数
	Fn02()

	// 函数打印数
	Fn03(5)

	// 递归函数执行1-100的和
	fmt.Printf("使用递归函数(方法)实现1-100的和:%v\n", Fn04(100))

	// 递归函数实现5的阶乘
	fmt.Printf("使用递归函数(方法)实现5的阶乘:%v\n", Fn05(5))

}
// 输出结果
/*
使用for循环实现1-100的和:5050
方法01
方法02
4
3
2
1
使用递归函数(方法)实现1-100的和:5050
使用递归函数(方法)实现5的阶乘:120
*/
```



### Go语言中的闭包



```go
Go语言中的闭包
全局变量特点：
  1. 常驻内存
  2. 污染全局
局部变量的特点：
  1. 不常住内存
  2. 不污染全局
闭包：
  1. 可以让一个变量常驻内存
  2. 可以让一个变量不污染全局
闭包概念：
  1. 闭包是只有权访问另一个函数作用域中的变量的函数；
  2. 创建闭包的常见方式就是在一个函数内部创建另一个函数，通过另一个函数访问这个函数的局部变量；
注意：
  由于闭包里作用域返回的局部变量资源不会被立刻销毁回收，所以可能会占用更多的内存，过渡使用闭包会导致性能下降，建议在非常有必要的时候才使用闭包；
```



### Go语言中的闭包实例



```go
package main

import "fmt"

// 定义一个全局变量
var A = "我是一个全局变量"

// 定义一个函数
func test() {
	// 定义一个局部变量(只在函数内部使用)
	b := "我是一个局部变量"
	fmt.Printf("输出全局变量的值:%v\n", A)
	fmt.Printf("输出局部变量的值:%v\n", b)
	// 修改全局变量
	var A = "全局变量被修改"
	fmt.Printf("输出全局变量(在函数内部被修改)的值:%v\n", A)
}

// 定义一个最简单的闭包

func adder1() func() int {
	var i = 88
	return func() int {
		return i + 1
	}
}

// 闭包的写法 函数里面嵌套一个函数，最后返回里面的函数
func adder2() func(y int) int {
	var i = 88
	return func(y int) int {
		i += y
		return i
	}
}

func main() {
	test()
	fmt.Printf("在main函数(方法)中输出全局变量A的值:%v\n", A)

	// 执行函数(方法)adder1
	var fn1 = adder1()
	fmt.Printf("fn的Type = %T   value = %v\n", fn1, fn1)
	fmt.Printf("输出fn的值:%v\n", fn1())

	var fn2 = adder2()
	fmt.Println(fn2(10))
	fmt.Println(fn2(10))
	fmt.Println(fn2(10))

}
// 输出结果
/*
输出全局变量的值:我是一个全局变量
输出局部变量的值:我是一个局部变量
输出全局变量(在函数内部被修改)的值:全局变量被修改   
在main函数(方法)中输出全局变量A的值:我是一个全局变量
fn的Type = func() int   value = 0x32b4e0
输出fn的值:89
98
108
118
*/
```



### Go语言中闭包函数递增



```go
package main

import "fmt"

// 定义一个闭包函数 需要传入参数
func incrementer() func(n int) int {
	i := 1
	return func(n int) int {
		i++
		return i
	}
}

// 定义一个闭包函数 不需要出入参数
func closure() func() int {
	var i = 0
	return func() int {
		i++
		return i
	}
}

func main() {
	name := "chadwick"

	func() {
		fmt.Printf("匿名自执行函数使用局部变量:%v\n", name)
	}()

	var inc = incrementer()
	fmt.Println(inc(1))
	fmt.Println(inc(1))
	fmt.Println(inc(1))

	clo := closure()
	fmt.Println(clo())
	fmt.Println(clo())
	fmt.Println(clo())
}
// 输出结果
/*
匿名自执行函数使用局部变量:chadwick
2
3
4
1
2
3
*/
```
